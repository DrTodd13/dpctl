56  static QVec* init_queues (backend BE, info::device_type DTy)
57  {
58  QVec *queues = new QVec();
59  auto Platforms = platform::get_platforms();
60  for (auto &p : Platforms) {
61  if (p.is_host()) continue;
62  auto be = p.get_backend();
63  auto Devices = p.get_devices();
64 
65  if (Devices.size() == 1) {
66  auto d = Devices[0];
67  auto devty = d.get_info<info::device::device_type>();
68  if(devty == DTy && be == BE) {
69  auto Ctx = context(d);
70  queues->emplace_back(Ctx, d);
71  break;
72  }
73  }
74  else {
75  vector_class<device> SelectedDevices;
76  for(auto &d : Devices) {
77  auto devty = d.get_info<info::device::device_type>();
78  if(devty == DTy && be == BE) {
79  SelectedDevices.push_back(d);
80 
81  // Workaround for situations when in some environments
82  // get_devices() returns each device TWICE. Then it fails in call
83  // for context constructor with all doubled devices.
84  // So use only one first device.
85  break;
86  }
87  }
88  if (SelectedDevices.size() > 0) {
89  auto Ctx = context(SelectedDevices);
90  auto d = SelectedDevices[0];
91  queues->emplace_back(Ctx, d);
92  }
93  }
94  }
95  return queues;
96  }

98  static QVec* init_active_queues ()
99  {
100  QVec *active_queues;
101  try {
102  auto def_device = std::move(default_selector().select_device());
103  auto BE = def_device.get_platform().get_backend();
104  auto DevTy = def_device.get_info<info::device::device_type>();
105 
106  // \todo : We need to have a better way to match the default device
107  // to what SYCL returns based on the same scoring logic. Just
108  // storing the first device is not correct when we will have
109  // multiple devices of same type.
110  if(BE == backend::opencl &&
111  DevTy == info::device_type::cpu) {
112  active_queues = new QVec({get_opencl_cpu_queues()[0]});
113  }
114  else if(BE == backend::opencl &&
115  DevTy == info::device_type::gpu) {
116  active_queues = new QVec({get_opencl_gpu_queues()[0]});
117  }
118  else if(BE == backend::level_zero &&
119  DevTy == info::device_type::gpu) {
120  active_queues = new QVec({get_level0_gpu_queues()[0]});
121  }
122  else {
123  active_queues = new QVec();
124  }
125  }
126  catch (runtime_error &re) {
127  // \todo Handle the error
128  active_queues = new QVec();
129  }
130 
131  return active_queues;
132  }

192 DPPLSyclQueueRef QMgrHelper::getCurrentQueue ()
193 {
194  auto &activated_q = get_active_queues();
195  if(activated_q.empty()) {
196  // \todo handle error
197  std::cerr << "No currently active queues.\n";
198  return nullptr;
199  }
200  auto last = activated_q.size() - 1;
201  return wrap(new queue(activated_q[last]));
202 }

210 __dppl_give DPPLSyclQueueRef
211 QMgrHelper::getQueue (DPPLSyclBackendType BETy,
212  DPPLSyclDeviceType DeviceTy,
213  size_t DNum)
214 {
215  queue *QRef = nullptr;
216 
217  switch (BETy|DeviceTy)
218  {
219  case DPPLSyclBackendType::DPPL_OPENCL | DPPLSyclDeviceType::DPPL_CPU:
220  {
221  auto cpuQs = get_opencl_cpu_queues();
222  if (DNum >= cpuQs.size()) {
223  // \todo handle error
224  std::cerr << "OpenCL CPU device " << DNum
225  << " not found on system.\n";
226  return nullptr;
227  }
228  QRef = new queue(cpuQs[DNum]);
229  break;
230  }
231  case DPPLSyclBackendType::DPPL_OPENCL | DPPLSyclDeviceType::DPPL_GPU:
232  {
233  auto gpuQs = get_opencl_gpu_queues();
234  if (DNum >= gpuQs.size()) {
235  // \todo handle error
236  std::cerr << "OpenCL GPU device " << DNum
237  << " not found on system.\n";
238  return nullptr;
239  }
240  QRef = new queue(gpuQs[DNum]);
241  break;
242  }
243  case DPPLSyclBackendType::DPPL_LEVEL_ZERO | DPPLSyclDeviceType::DPPL_GPU:
244  {
245  auto l0GpuQs = get_level0_gpu_queues();
246  if (DNum >= l0GpuQs.size()) {
247  // \todo handle error
248  std::cerr << "Level-0 GPU device " << DNum
249  << " not found on system.\n";
250  return nullptr;
251  }
252  QRef = new queue(l0GpuQs[DNum]);
253  break;
254  }
255  default:
256  std::cerr << "Unsupported device type.\n";
257  return nullptr;
258  }
259 
260  return wrap(QRef);
261 }

268 bool QMgrHelper::isCurrentQueue (__dppl_keep const DPPLSyclQueueRef QRef)
269 {
270  auto &activated_q = get_active_queues();
271  if(activated_q.empty()) {
272  // \todo handle error
273  std::cerr << "No currently active queues.\n";
274  return false;
275  }
276  auto last = activated_q.size() - 1;

278  return (*unwrap(QRef) == currQ);

352 __dppl_give DPPLSyclQueueRef
353 QMgrHelper::pushSyclQueue (DPPLSyclBackendType BETy,
354  DPPLSyclDeviceType DeviceTy,
355  size_t DNum)
356 {
357  queue *QRef = nullptr;
358  auto &activeQ = get_active_queues();
359  if(activeQ.empty()) {
360  std::cerr << "Why is there no previous global context?\n";
361  return nullptr;
362  }
363 
364  switch (BETy|DeviceTy)
365  {
366  case DPPLSyclBackendType::DPPL_OPENCL | DPPLSyclDeviceType::DPPL_CPU:
367  {
368  if (DNum >= get_opencl_cpu_queues().size()) {
369  // \todo handle error
370  std::cerr << "OpenCL CPU device " << DNum
371  << " not found on system\n.";
372  return nullptr;
373  }
374  activeQ.emplace_back(get_opencl_cpu_queues()[DNum]);
375  QRef = new queue(activeQ[activeQ.size()-1]);
376  break;
377  }
378  case DPPLSyclBackendType::DPPL_OPENCL | DPPLSyclDeviceType::DPPL_GPU:
379  {
380  if (DNum >= get_opencl_gpu_queues().size()) {
381  // \todo handle error
382  std::cerr << "OpenCL GPU device " << DNum
383  << " not found on system\n.";
384  return nullptr;
385  }
386  activeQ.emplace_back(get_opencl_gpu_queues()[DNum]);
387  QRef = new queue(activeQ[get_active_queues().size()-1]);
388  break;
389  }
390  case DPPLSyclBackendType::DPPL_LEVEL_ZERO | DPPLSyclDeviceType::DPPL_GPU:
391  {
392  if (DNum >= get_level0_gpu_queues().size()) {
393  // \todo handle error
394  std::cerr << "Level-0 GPU device " << DNum
395  << " not found on system\n.";
396  return nullptr;
397  }
398  activeQ.emplace_back(get_level0_gpu_queues()[DNum]);
399  QRef = new queue(activeQ[get_active_queues().size()-1]);
400  break;
401  }
402  default:
403  {
404  std::cerr << "Unsupported device type.\n";
405  return nullptr;
406  }
407  }
408 
409  return wrap(QRef);
410 }
