167 DPPLQueue_SubmitRange (__dppl_keep const DPPLSyclKernelRef KRef,
168  __dppl_keep const DPPLSyclQueueRef QRef,
169  __dppl_keep void **Args,
170  __dppl_keep const DPPLKernelArgType *ArgTypes,
171  size_t NArgs,
172  __dppl_keep const size_t Range[3],
173  size_t NDims,
174  __dppl_keep const DPPLSyclEventRef *DepEvents,
175  size_t NDepEvents)
176 {
177  auto Kernel = unwrap(KRef);
178  auto Queue = unwrap(QRef);
179  event e;
180 
181  try {
182  e = Queue->submit([&](handler& cgh) {
183  // Depend on any event that was specified by the caller.
184  if(NDepEvents)
185  for(auto i = 0ul; i < NDepEvents; ++i)
186  cgh.depends_on(*unwrap(DepEvents[i]));
187 
188  for (auto i = 0ul; i < NArgs; ++i) {
189  // \todo add support for Sycl buffers
190  // \todo handle errors properly
191  if(!set_kernel_arg(cgh, i, Args[i], ArgTypes[i]))
192  exit(1);
193  }
194  switch(NDims)
195  {
196  case 1:
197  cgh.parallel_for(range<1>{Range[0]}, *Kernel);
198  break;
199  case 2:
200  cgh.parallel_for(range<2>{Range[0], Range[1]}, *Kernel);
201  break;
202  case 3:
203  cgh.parallel_for(range<3>{Range[0], Range[1], Range[2]},
204  *Kernel);
205  break;
206  default:
207  // \todo handle the error
208  throw std::runtime_error("Range cannot be greater than three "
209  "dimensions.");
210  }
211  });
212  } catch (runtime_error &re) {
213  // \todo fix error handling
214  std::cerr << re.what() << '\n';
215  return nullptr;
216  } catch (std::runtime_error &sre) {
217  std::cerr << sre.what() << '\n';
218  return nullptr;
219  }
220 
221  return wrap(new event(e));
222 }
