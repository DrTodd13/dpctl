249 static int get_first_device (cl_platform_id* platforms,
250  cl_uint platformCount,
251  cl_device_id *device,
252  cl_device_type device_ty)
253 {
254  cl_int status;
255  cl_uint ndevices = 0;
256  unsigned int i;
257 
258  for (i = 0; i < platformCount; ++i) {
259  // get all devices of env_ty
260  status = clGetDeviceIDs(platforms[i], device_ty, 0, NULL, &ndevices);
261  // If this platform has no devices of this type then continue
262  if(!ndevices) continue;
263 
264  // get the first device
265  status = clGetDeviceIDs(platforms[i], device_ty, 1, device, NULL);
266  CHECK_OPEN_CL_ERROR(status, "Could not get first cl_device_id.");
267 
268  // If the first device of this type was discovered, no need to look more
269  if(ndevices) break;
270  }
271 
272  if(ndevices)
273  return DP_GLUE_SUCCESS;
274  else
275  return DP_GLUE_FAILURE;
276 
277 error:
278  return DP_GLUE_FAILURE;
279 }

328 static int create_dp_env_t (cl_platform_id* platforms,
329  size_t nplatforms,
330  cl_device_type device_ty,
331  env_t *env_t_ptr)
332 {
333  cl_int err;
334  int err1;
335  env_t env;
336  cl_device_id *device;
337 
338  env = NULL;
339  device = NULL;
340 
341  // Allocate the env_t object
342  env = (env_t)malloc(sizeof(struct dp_env));
343  CHECK_MALLOC_ERROR(env_t, env);
344  env->id_ = ENV_ID;
345 
346  env->context = NULL;
347  env->device = NULL;
348  env->queue = NULL;
349  env->max_work_item_dims = 0;
350  env->max_work_group_size = 0;
351  env->dump_fn = NULL;
352 
353  device = (cl_device_id*)malloc(sizeof(cl_device_id));
354 
355  err1 = get_first_device(platforms, nplatforms, device, device_ty);
356  CHECK_DPGLUE_ERROR(err1, "Failed inside get_first_device");
357 
358  // get the CL_DEVICE_MAX_WORK_ITEM_DIMENSIONS for this device
359  err = clGetDeviceInfo(*device, CL_DEVICE_MAX_WORK_ITEM_DIMENSIONS,
360  sizeof(env->max_work_item_dims), &env->max_work_item_dims, NULL);
361  CHECK_OPEN_CL_ERROR(err, "Could not get max work item dims");
362 
363  // get the CL_DEVICE_MAX_WORK_GROUP_SIZE for this device
364  err = clGetDeviceInfo(*device, CL_DEVICE_MAX_WORK_GROUP_SIZE,
365  sizeof(env->max_work_group_size), &env->max_work_group_size, NULL);
366  CHECK_OPEN_CL_ERROR(err, "Could not get max work group size");
367 
368  // Create a context and associate it with device
369  env->context = clCreateContext(NULL, 1, device, NULL, NULL, &err);
370  CHECK_OPEN_CL_ERROR(err, "Could not create device context.");
371  // Create a queue and associate it with the context
372  env->queue = clCreateCommandQueueWithProperties((cl_context)env->context,
373  *device, 0, &err);
374 
375  CHECK_OPEN_CL_ERROR(err, "Could not create command queue.");
376 
377  env->device = *device;
378  env ->dump_fn = dump_device_info;
379 
380  if (DP_GLUE_SUCCESS == support_int64_atomics(device)) {
381  env->support_int64_atomics = 1;
382  } else {
383  env->support_int64_atomics = 0;
384  }
385 
386  if (DP_GLUE_SUCCESS == support_float64_atomics(device)) {
387  env->support_float64_atomics = 1;
388  } else {
389  env->support_float64_atomics = 0;
390  }
391 
392  free(device);
393  *env_t_ptr = env;
394 
395  return DP_GLUE_SUCCESS;
396 
397 malloc_error:
398  return DP_GLUE_FAILURE;
399 error:
400  free(env);
401  *env_t_ptr = NULL;
402  return DP_GLUE_FAILURE;
403 }

431 static int init_runtime_t_obj (runtime_t rt)
432 {
433  cl_int status;
434  int ret;
435  cl_platform_id *platforms;
436 #if DEBUG
437  check_runtime_id(rt);
438 #endif
439  // get count of available platforms
440  status = clGetPlatformIDs(0, NULL, &(rt->num_platforms));
441  CHECK_OPEN_CL_ERROR(status, "Could not get platform count.");
442 
443  if(!rt->num_platforms) {
444  fprintf(stderr, "No OpenCL platforms found.\n");
445  goto error;
446  }
447 
448  // Allocate memory for the platforms array
449  rt->platform_ids = (cl_platform_id*)malloc(
450  sizeof(cl_platform_id)*rt->num_platforms
451  );
452  CHECK_MALLOC_ERROR(cl_platform_id, rt->platform_ids);
453 
454  // Get the platforms
455  status = clGetPlatformIDs(rt->num_platforms, rt->platform_ids, NULL);
456  CHECK_OPEN_CL_ERROR(status, "Could not get platform ids");
457  // Cast rt->platforms to a pointer of type cl_platform_id, as we cannot do
458  // pointer arithmetic on void*.
459  platforms = (cl_platform_id*)rt->platform_ids;
460  // Get the first cpu device on this platform
461  ret = create_dp_env_t(platforms, rt->num_platforms,
462  CL_DEVICE_TYPE_CPU, &rt->first_cpu_env);
463  rt->has_cpu = !ret;
464 
465 #if DEBUG
466  if(rt->has_cpu)
467  printf("DEBUG: CPU device acquired...\n");
468  else
469  printf("DEBUG: No CPU available on the system\n");
470 #endif
471 
472  // Get the first gpu device on this platform
473  ret = create_dp_env_t(platforms, rt->num_platforms,
474  CL_DEVICE_TYPE_GPU, &rt->first_gpu_env);
475  rt->has_gpu = !ret;
476 
477 #if DEBUG
478  if(rt->has_gpu)
479  printf("DEBUG: GPU device acquired...\n");
480  else
481  printf("DEBUG: No GPU available on the system.\n");
482 #endif
483 
484  if(rt->has_gpu)
485  rt->curr_env = rt->first_gpu_env;
486  else if(rt->has_cpu)
487  rt->curr_env = rt->first_cpu_env;
488  else
489  goto error;
490 
491  return DP_GLUE_SUCCESS;
492 
493 malloc_error:
494 
495  return DP_GLUE_FAILURE;
496 error:
497  free(rt->platform_ids);
498 
499  return DP_GLUE_FAILURE;
500 }

728  CHECK_OPEN_CL_ERROR(err, "Failed to write to CL buffer.");
729 
730  err = clReleaseCommandQueue(queue);
731  CHECK_OPEN_CL_ERROR(err, "Failed to release the command queue.");
732  err = clReleaseMemObject(mem);
733  CHECK_OPEN_CL_ERROR(err, "Failed to release the buffer memory object.");
734 
735 #if DEBUG
736  printf("DEBUG: CL buffer written to device...\n");
737 #endif
738  //--- TODO: Implement a version that uses clEnqueueMapBuffer
739 
740  return DP_GLUE_SUCCESS;
741 error:
