281 TEST_F (TestDPPLSyclQueueInterface, CheckSubmit)
282 {
283  if(!has_devices())
284  GTEST_SKIP_("Skipping: No Sycl devices.\n");
285 
286  auto nOpenCLGpuQ = DPPLQueueMgr_GetNumQueues(DPPL_OPENCL, DPPL_GPU);
287 
288  if(!nOpenCLGpuQ)
289  GTEST_SKIP_("Skipping: No OpenCL GPU device.\n");
290 
291  auto Queue = DPPLQueueMgr_GetQueue(DPPL_OPENCL, DPPL_GPU, 0);
292  auto CtxRef = DPPLQueue_GetContext(Queue);
293  auto PRef = DPPLProgram_CreateFromOCLSource(CtxRef, CLProgramStr,
294  CompileOpts);
295  ASSERT_TRUE(PRef != nullptr);
296  ASSERT_TRUE(DPPLProgram_HasKernel(PRef, "init_arr"));
297  ASSERT_TRUE(DPPLProgram_HasKernel(PRef, "add"));
298  ASSERT_TRUE(DPPLProgram_HasKernel(PRef, "axpy"));
299 
300  auto InitKernel = DPPLProgram_GetKernel(PRef, "init_arr");
301  auto AddKernel = DPPLProgram_GetKernel(PRef, "add");
302  auto AxpyKernel = DPPLProgram_GetKernel(PRef, "axpy");
303 
304  // Create the input args
305  auto a = DPPLmalloc_shared(SIZE, Queue);
306  ASSERT_TRUE(a != nullptr);
307  auto b = DPPLmalloc_shared(SIZE, Queue);
308  ASSERT_TRUE(b != nullptr);
309  auto c = DPPLmalloc_shared(SIZE, Queue);
310  ASSERT_TRUE(c != nullptr);
311 
312  // Initialize a,b
313  DPPLKernelArgType argTypes[] = {DPPL_VOID_PTR};
314  size_t Range[] = {SIZE};
315  void *arg1[1] = { unwrap(a) };
316  void *arg2[1] = { unwrap(b) };
317 
318  auto E1 = DPPLQueue_SubmitRange(InitKernel, Queue, arg1, argTypes, 1,
319  Range, 1, nullptr, 0);
320  auto E2 = DPPLQueue_SubmitRange(InitKernel, Queue, arg2, argTypes, 1,
321  Range, 1, nullptr, 0);
322  ASSERT_TRUE(E1 != nullptr);
323  ASSERT_TRUE(E2 != nullptr);
324 
325  DPPLQueue_Wait(Queue);
326 
327  // Submit the add kernel
328  void *args[3] = { unwrap(a), unwrap(b), unwrap(c) };
329  DPPLKernelArgType addKernelArgTypes[] = {
330  DPPL_VOID_PTR,
331  DPPL_VOID_PTR,
332  DPPL_VOID_PTR
333  };
334 
335  auto E3 = DPPLQueue_SubmitRange(AddKernel, Queue, args,
336  addKernelArgTypes, 3, Range, 1, nullptr, 0);
337  ASSERT_TRUE(E3 != nullptr);
338  DPPLQueue_Wait(Queue);
339 
340  // Verify the result of "add"
341  add_kernel_checker((float*)a, (float*)b, (float*)c);
342 
343  // Create kernel args for axpy
344  float d = 10.0;
345  void *args2[4] = { unwrap(a), unwrap(b), unwrap(c) , (void*)&d };
346  DPPLKernelArgType addKernelArgTypes2[] = {
347  DPPL_VOID_PTR,
348  DPPL_VOID_PTR,
349  DPPL_VOID_PTR,
350  DPPL_FLOAT
351  };
352  auto E4 = DPPLQueue_SubmitRange(AxpyKernel, Queue, args2,
353  addKernelArgTypes2, 4, Range, 1,
354  nullptr, 0);
355  ASSERT_TRUE(E4 != nullptr);
356  DPPLQueue_Wait(Queue);
357 
358  // Verify the result of "axpy"
359  axpy_kernel_checker((float*)a, (float*)b, (float*)c, d);
360 
361  // clean ups
362  DPPLEvent_Delete(E1);
363  DPPLEvent_Delete(E2);
364  DPPLEvent_Delete(E3);
365  DPPLEvent_Delete(E4);
366 
367  DPPLKernel_Delete(AddKernel);
368  DPPLKernel_Delete(AxpyKernel);
369  DPPLKernel_Delete(InitKernel);
370 
371  DPPLfree_with_queue((DPPLSyclUSMRef)a, Queue);
372  DPPLfree_with_queue((DPPLSyclUSMRef)b, Queue);
373  DPPLfree_with_queue((DPPLSyclUSMRef)c, Queue);
374 
375  DPPLQueue_Delete(Queue);
376  DPPLContext_Delete(CtxRef);
377  DPPLProgram_Delete(PRef);
