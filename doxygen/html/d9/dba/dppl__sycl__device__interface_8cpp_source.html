45 void dump_device_info (const device & Device)
46 {
47  std::stringstream ss;
48 
49  ss << std::setw(4) << " " << std::left << std::setw(16) << "Name"
50  << Device.get_info<info::device::name>() << '\n';
51  ss << std::setw(4) << " " << std::left << std::setw(16) << "Driver version"
52  << Device.get_info<info::device::driver_version>() << '\n';
53  ss << std::setw(4) << " " << std::left << std::setw(16) << "Vendor"
54  << Device.get_info<info::device::vendor>() << '\n';
55  ss << std::setw(4) << " " << std::left << std::setw(16) << "Profile"
56  << Device.get_info<info::device::profile>() << '\n';
57  ss << std::setw(4) << " " << std::left << std::setw(16) << "Device type";
58 
59  auto devTy = Device.get_info<info::device::device_type>();
60  switch(devTy)
61  {
62  case info::device_type::cpu:
63  ss << "cpu" << '\n';
64  break;
65  case info::device_type::gpu:
66  ss << "gpu" << '\n';
67  break;
68  case info::device_type::accelerator:
69  ss << "accelerator" << '\n';
70  break;
71  case info::device_type::custom:
72  ss << "custom" << '\n';
73  break;
74  case info::device_type::host:
75  ss << "host" << '\n';
76  break;
77  default:
78  ss << "unknown" << '\n';
79  }
80 
81  std::cout << ss.str();
82 }
