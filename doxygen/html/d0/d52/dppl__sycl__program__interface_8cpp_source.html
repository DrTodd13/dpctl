57  auto CLProgram = clCreateProgramWithIL(CLCtx, IL, length, &err);
58  if (err) {
59  // \todo: record the error string and any other information.
60  std::cerr << "OpenCL program could not be created from the SPIR-V "

62  return nullptr;
63  }
64  auto SyclDevices = SyclCtx->get_devices();
65 
66  // Get a list of CL Devices from the Sycl devices
67  auto CLDevices = new cl_device_id[SyclDevices.size()];
68  for (auto i = 0ul; i < SyclDevices.size(); ++i)
69  CLDevices[i] = SyclDevices[i].get();

71  // Build the OpenCL interoperability program

73  nullptr, nullptr, nullptr);
74  // free the CLDevices array
75  delete[] CLDevices;
76

78  // \todo: record the error string and any other information.
79  std::cerr << "OpenCL program could not be built. OpenCL Error "
80  << err << ".\n";
81  return nullptr;
82  }
